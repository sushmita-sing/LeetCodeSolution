# def isBadVersion(version: int) -> bool:

class Solution:
    def firstBadVersion(self, n: int) -> int:
        l, r, mid = 0, n, 0
        while l<=r:
            mid = (l+r)//2
            if isBadVersion(mid):
                if not isBadVersion(mid-1): 
                    return mid
                else:
                    r = mid-1
            else:
                l = mid+1
